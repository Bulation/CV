Мое решение этой каты из кодварс

Задача: реализовать конструктор Cat, которые принимает в качестве аргументов имя и вес для создания объекта кота. Конструктор также должен иметь метод averageWeight, который возвращает средний вес котов, созданных в конструкторе.

Решение на JavaScript:


var Cat = (function () {
  let weights = {};
  function constructor (name, weight) {
    if (!name || !weight)
      throw new Error();
    this.name = name;
    this.weight = weight;
    Object.defineProperty(this, 'weight', {
      get: () => weight,
      set: value => {
        weight = value;
        weights[name] = weight;
      }
    })
    weights[name] = weight;
  }
  constructor.averageWeight = function () {
    let s = 0;
    for (let weight in weights) {
      s += weights[weight];
    }
    return s / Object.values(weights).length;
  }
  return constructor;
}());
