Code example

My solution of this codewars kata

Problem: Make a Cat constructor that takes arguments name and weight to instantiate a new cat object. The constructor should also have an averageWeight method that returns the average weight of cats created with the constructor.

Solution on JavaScript:


var Cat = (function () {
  let weights = {};
  function constructor (name, weight) {
    if (!name || !weight)
      throw new Error();
    this.name = name;
    this.weight = weight;
    Object.defineProperty(this, 'weight', {
      get: () => weight,
      set: value => {
        weight = value;
        weights[name] = weight;
      }
    })
    weights[name] = weight;
  }
  constructor.averageWeight = function () {
    let s = 0;
    for (let weight in weights) {
      s += weights[weight];
    }
    return s / Object.values(weights).length;
  }
  return constructor;
}());
